577 unsigned char* RijndaelKeySchedule(unsigned char* originalKey, int keyLength)
578 {
579  uint8_t currentNumBytes = (keyLength); // number of bytes of the key we have so far
580  uint8_t rConIter = 1; // counter for RCON
581  int numRounds = getNumRounds(keyLength); // number of rounds of the encryption
582 
583  validateNumRounds(numRounds,keyLength);
584 
585  // we need 128 bits * (number of rounds +1) key at the end
586  uint8_t bytesNeeded = ( (numRounds + 1) *AES_BLOCK_SIZE);
587  unsigned char* expandedKey = calloc(bytesNeeded, sizeof(char));
588  for(uint8_t x = 0; x < keyLength; x++) {
589  expandedKey[x] = originalKey[x];
590  }
591 
592  int tempWordLength = 4;
593  unsigned char tempArr[tempWordLength]; // temp vector used for the keyExpansion
594 
595  // while number of bytes of key > number of bytes needed
596  while(currentNumBytes < bytesNeeded) {
597  // create 4 bytes
598  for(uint8_t x = 0; x < 4; x++) {
599  tempArr[x] = expandedKey[(currentNumBytes - 4) + x];
600  }
601  /* every 16,24,32 bytes we apply the core schedule to t and increment rconIteration afterwards */
602  if(currentNumBytes % keyLength == 0){
603  KeyScheduleCore(tempArr, tempWordLength,rConIter);
604  rConIter++;
605  }
606 
607  /* For 256-bit keys, we add an extra sbox to the calculation */
608  if(keyLength == 32 && ((currentNumBytes % keyLength) == 16)) {
609  for(uint8_t x = 0; x < 4; x++) {
610  tempArr[x] = getSBoxValue(tempArr[x]);
611  }
612  }
613 
614  // We XOR tempArr with the four-byte block 16,24,32 bytes before the new expanded key.
615  // This becomes the next four bytes in the expanded key.
616  for(uint8_t x = 0; x < 4; x++) {
617  expandedKey[currentNumBytes] = expandedKey[currentNumBytes - keyLength] ^ tempArr[x]; //Bitwise XOR
618  currentNumBytes++;
619  }
620  }
621  return expandedKey;
622 }
